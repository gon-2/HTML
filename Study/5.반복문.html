반복문 loop : 동일한 작업을 여러번 반복

        1. for : 
        for(let i = 0; i < 10; i++){ 
            // let i = 0; << 초기값
            // i < 10; << 조건식
            // i++ << 증감식
            // 반복할 코드를 넣으면 됨
        }
    
        // 1부터 10까지 로그를 찍어보자
        for(let i = 0; i < 10; i++){ 
            // let i = 1; i < 11; i++ 과 같다
            // let i = 1; i <= 10 i++ 과 같다
            console.dir(i+1);
        }



       2. while
       let i = 0;

       while(i<10){
        console.dir(i);
        i++;  // i++을 해주는 이유는 증감을 해주지 않으면 무한 반복이 되기 때문 
       }

       3. do..while
       do{
        // 코드
        i++;
       }while(i<10)

       4. 반복문을 빠져나오는 기능을 알아보자!
       break, continue가 있다

       break : 멈추고 빠져나옴
       continue : 멈추고 다음 반복으로 진행


       // break
       while(true){
        let answer = confirm('계속할까여?');
            if(!answer){
                break;
            }
        }   

        // continue
        // 짝수만 불러오기
        for(let i = 0; i < 10; i++){
            if(i%2){
                continue;
            }
            console.dir(i);
        }
        
        명확한 횟수가 정해져 있으면 for문 그렇지 않다면 while문을 사용한다
